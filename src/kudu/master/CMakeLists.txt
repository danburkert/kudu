KRPC_GENERATE(MASTER_KRPC_SRCS MASTER_KRPC_HDRS
  SOURCE_ROOT ${CMAKE_CURRENT_SOURCE_DIR}/../..

  PROTO_FILES master.proto)

add_library(master_proto ${MASTER_KRPC_SRCS} ${MASTER_KRPC_HDRS})

  metadata_proto
  tablet_proto

  protobuf
  rpc_header_proto
  krpc)

set(MASTER_SRCS
  catalog_manager.cc
  master.cc
  master_options.cc
  master_service.cc
  master-path-handlers.cc
  mini_master.cc
  sys_tables.cc
  ts_descriptor.cc
  ts_manager.cc

  kudu_common
  tablet
  server_common
  server_process
  krpc
  gutil
  kudu_util
  tserver
  tserver_service_proto
  master_proto
  rpc_header_proto)

  master
  ${KUDU_BASE_LIBS})
